CREATE TABLE raksh_branch_tab (
    branch_id NUMBER GENERATED BY DEFAULT AS IDENTITY,
    company_reg_no NUMBER NOT NULL,
    branch_name VARCHAR2(200) NOT NULL,
    branch_address VARCHAR2(4000) NOT NULL,
    branch_manager VARCHAR2(200) NOT NULL,
    total_warehouse_capacity NUMBER NOT NULL,
    PRIMARY KEY (company_reg_no, branch_id,)
    
);
